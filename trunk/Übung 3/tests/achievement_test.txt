REM curl test script of UGNM service.
REM
REM Tests resources /achievements and /achievements/{id} including all supported HTTP operations
REM and all possible responses. Unsupported operations are not explicitly tested.

REM get all achievements
curl http://localhost:8080/ugnm-service/resources/achievements/ -v -X GET

REM create new achievements with id "004"
curl -v -X PUT http://localhost:8080/ugnm-service/resources/achievements/ -H "Content-Type: application/json" -d "{'id':004,'description':'Neues Test-Achievement','name':'Achievement4','url':'/achievements/004'}"

REM try to create new achievements with id "004" again. Should result in conflict.
curl -v -X PUT http://localhost:8080/ugnm-service/resources/achievements/ -H "Content-Type: application/json" -d "{'id':004,'description':'Neues Test-Achievement','name':'Achievement4','url':'/achievements/004'}"

REM try to create new achievements with missing parameter.
curl http://localhost:8080/ugnm-service/resources/achievements -v -X PUT -H "Content-Type: application/json" -d "{'id':005}"

REM get non-existing achievements
curl http://localhost:8080/ugnm-service/resources/achievements/idonotexist -v -X GET

REM get data of achievements "004"
curl http://localhost:8080/ugnm-service/resources/achievements/004 -v -X GET

REM update data of achievements "004" without data.
curl http://localhost:8080/ugnm-service/resources/achievements/004 -v -X PUT -H "Content-Type: application/json" -d "{}"

REM update data of achievements "004"
curl http://localhost:8080/ugnm-service/resources/achievements/004 -v -X PUT -H "Content-Type: application/json" -d "{'id':004,'description':'Neues Test-Achievement Update','name':'Achievement4','url':'/achievements/004Upd'}"

REM get updated data of achievements "004"
curl http://localhost:8080/ugnm-service/resources/achievements/004 -v -X GET

REM update data of non-existing achievements
curl http://localhost:8080/ugnm-service/resources/achievements/000 -v -X PUT -H "Content-Type: application/json" -d "{'id':000,'description':'not exist','name':'idnotexist','url':'/achievements/004nex'}"

REM delete achievements "004" - success
curl http://localhost:8080/ugnm-service/resources/achievements/004 -v -X DELETE

REM delete non-existing achievements "004" (was deleted before)
curl http://localhost:8080/ugnm-service/resources/achievements/004 -v -X DELETE
