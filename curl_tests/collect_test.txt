REM curl test script of UGNM service.
REM
REM Tests resources /achievements and /achievements/{id} including all supported HTTP operations
REM and all possible responses. Unsupported operations are not explicitly tested.

REM get all collected Achievments of user "sven.hausburg@rwth-aachen.de"
curl -v -X GET http://localhost:8080/ugnm-service/resources/users/sven.hausburg@rwth-aachen.de/collect

REM create new collect "sven.hausburg" to "achievment 4"
curl -v -X PUT http://localhost:8080/ugnm-service/resources/users/sven.hausburg@rwth-aachen.de/collect -H "Content-Type: application/json" -d "{'achievementId':4}" --user sven.hausburg@rwth-aachen.de:abc123

REM create new collect "sven.hausburg" to "achievment 4" without authorization
curl -v -X PUT http://localhost:8080/ugnm-service/resources/users/sven.hausburg@rwth-aachen.de/collect -H "Content-Type: application/json" -d "{'achievementId':4}"

REM try to create new rating with missing parameter
curl -v -X PUT http://localhost:8080/ugnm-service/resources/users/sven.hausburg@rwth-aachen.de/collect -H "Content-Type: application/json" -d "{'notexist':'nx'}"

REM try to collect non-existing achievement
curl -v -X PUT http://localhost:8080/ugnm-service/resources/users/sven.hausburg@rwth-aachen.de/collect -H "Content-Type: application/json" -d "{'achievementId':77}" --user sven.hausburg@rwth-aachen.de:abc123

REM get data of collect by Achievment ID
curl -v -X GET http://localhost:8080/ugnm-service/resources/users/sven.hausburg@rwth-aachen.de/collect/4
